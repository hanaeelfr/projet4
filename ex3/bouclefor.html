Afficher la table de multiplication d’un nombre saisi par l’utilisateur dans un input.

document.getElementById('button').onclick=function(){
var i,resultat="";
var x = document.getElementById('number').value ;
for( i=1; i<=10; i++){
resultat=resultat+"
"+x+"x"+i+"="+i*x+";

}
}
